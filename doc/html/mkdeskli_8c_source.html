149 void parsIncomingArg(int argc, char* argv[])
150 {
151 
152  int opt;
153  int option_index = 0;
154  static struct option long_options[] =
155  {
156  {"version", no_argument, 0, 'v'},
157  {"help", no_argument, 0, 'h'},
158  {"name", required_argument, 0, 'n'},
159  {"path", required_argument, 0, 'p'},
160  {"image", required_argument, 0, 'i'},
161  {"use-editor", no_argument, 0, 'u'},
162  {"user-only", no_argument, 0, 'o'},
163  {0, 0, 0, 0}
164  };
165 
166  static char *options_string = "vhn:p:i:uo";
167 
168  while ((opt = getopt_long(argc, argv, options_string, long_options, &option_index)) != -1)
169  {
170  switch (opt)
171  {
172  case 'v':
173  printf(APP_NAME " version: " APP_VERSION ".\n");
174  exit(EXIT_SUCCESS);
175  break;
176  case 'h':
177  phelp(argv[0]);
178  exit(EXIT_SUCCESS);
179  break;
180  case 'n':
181  setFlag(NAME);
182  if (optarg)
183  {
184  strcpy(desklink.name, optarg);
185  //strcat(desklink.name, FILE_TYPE);
186  }
187  break;
188  case 'p':
189  setFlag(PATH);
190  if (optarg)
191  {
192  strcpy(desklink.appPath, optarg);
193  }
194  break;
195  case 'i':
196  setFlag(IMAGE);
197  if (optarg)
198  {
199  strcpy(desklink.imagePath, optarg);
200  }
201  break;
202  case 'u':
203  setFlag(EDITOR);
204  break;
205  case 'o':
206  setFlag(UONLY);
207  break;
208 
209  case '?':
210  default:
211  exit(EXIT_FAILURE);
212 
213  }//switch (opt)
214  }//while
215 }
