221 void parsIncomingArg(int argc, char* argv[])
222 {
223 
224  int opt;
225  int option_index = 0;
226  static struct option long_options[] =
227  {
228  {"version", no_argument, 0, 'v'},
229  {"help", no_argument, 0, 'h'},
230  {"name", required_argument, 0, 'n'},
231  {"path", required_argument, 0, 'p'},
232  {"image", required_argument, 0, 'i'},
233  {"all-users", no_argument, 0, 'a'},
234  {"user-only", no_argument, 0, 'o'},
235  {"terminal", no_argument, 0, 't'},
236  {"remove", required_argument, 0, 'r'},
237  {"search", required_argument, 0, 's'},
238  {0, 0, 0, 0}
239  };
240 
241  static char *options_string = "vhn:p:i:uatr:s:";
242 
243  while ((opt = getopt_long(argc, argv, options_string, long_options, &option_index)) != -1)
244  {
245  switch (opt)
246  {
247  case 'v':
248  printf(APP_NAME " version: " APP_VERSION ".\n");
249  exit(EXIT_SUCCESS);
250  break;
251  case 'h':
252  phelp(argv[0]);
253  exit(EXIT_SUCCESS);
254  break;
255  case 'n':
256  if (optarg)
257  {
258  setFlag(NAME);
259  strcpy(desklink.name, optarg);
260 
261  }
262  break;
263  case 'p':
264 
265  if (optarg)
266  {
267  setFlag(PATH);
268  strcpy(desklink.appPath, optarg);
269  strcat(desklink.appPath, "\n");
270  }
271  break;
272  case 'i':
273 
274  if (optarg)
275  {
276  setFlag(IMAGE);
277  strcpy(desklink.imagePath, optarg);
278 
279  }
280  break;
281  case 'u':
282  setFlag(EDITOR);
283  break;
284  case 'a':
285  setFlag(ALLU);
286  break;
287  case 't':
288  setFlag(TERM);
289  desklink.terminal = 1;
290  break;
291  case 'r':
292  if (optarg)
293  {
294  setFlag(REMV);
295  strcpy(desklink.name, optarg);
296  }
297  break;
298  case 's':
299  if (optarg)

301  setFlag(SRCH);
302  strcpy(desklink.name, optarg);
303 
304  }
305  break;
306  case '?':

310  }//switch (opt)
